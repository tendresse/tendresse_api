# API for the Tendresse application, deployed on Heroku
API Stack : Flask + socket.io + SQLAlchemy + Postgres
  
when deployed, set the env_vars :  
'SECRET_KEY'  
'IONIC_API_TOKEN'  
'DATABASE_URI'  
  
useful ressources :
